使用Bootstrap和jQuery制作

点击弹框 
$( "#popup" ).click(function( event ) {
         alert("触发成功！");
});
点击消失 
$( "#disappear" ).click(function( event ) {
	$( this ).hide( "slow" );
});
点击变淡 
$( "#fadeout" ).click(function( event ) {
	$( this ).animate({opacity: 0.5});
});
点我重置 
$( "#reset" ).click(function( event ) {
	$( "#disappear" ).show( "slow" );
	$( "#fadeout" ).animate({opacity: 1});
});
